# lm-tool Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build `lm`, a standalone cross-platform DevOps CLI that supports `lm init`, `lm build server|web|admin`, `lm build`, and `lm help` for the lumu99 repositories.

**Architecture:** Use a small TypeScript CLI with a thin command parser, a central help registry, and testable core modules for config, path handling, init planning, command execution, jar discovery, and platform-specific server restart logic. Keep interactive prompt code isolated from business logic so the init and build flows can be tested with fake prompt, filesystem, and executor adapters.

**Tech Stack:** Node.js, TypeScript, Vitest, `@inquirer/prompts`, built-in `child_process`, `fs`, and `path`, plus `pkg` for standalone executable packaging.

export function createConfigStore(deps: ConfigDeps) {
  const configPath = join(deps.executableDir, 'lm.config.json');

  return {
    getPath: () => configPath,
    async load() {
      if (!(await deps.fs.exists(configPath))) {
        return null;
      }
      return JSON.parse(await deps.fs.readFile(configPath));
    },
    async save(config: LmConfig) {
      await deps.fs.writeFile(configPath, JSON.stringify(config, null, 2));
    },
  };
}

export function normalizeProjectPath(input: NormalizeInput): NormalizeResult {
  const pathApi = input.platform === 'windows' ? win32 : posix;
  const normalized = pathApi.normalize(input.input.replaceAll('\\', pathApi.sep));
  const absolute = pathApi.isAbsolute(normalized) ? normalized : pathApi.resolve(input.cwd, normalized);

  return {
    original: input.input,
    normalized: absolute,
    isAbsolute: pathApi.isAbsolute(absolute),
  };
}

export function createExecutor(deps: ExecutorDeps) {
  return {
    async run(input: RunProcessInput): Promise<RunProcessResult> {
      const child = deps.spawnImpl(input.command, input.args, {
        cwd: input.cwd,
        shell: false,
      });

      child.stdout?.on('data', (chunk) => input.onStdout?.(String(chunk)));
      child.stderr?.on('data', (chunk) => input.onStderr?.(String(chunk)));
